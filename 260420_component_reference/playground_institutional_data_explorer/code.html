SELECT
  f.region_code,
  f.facility_type,
  SUM(f.total_sqft) AS current_sqft,
  SUM(f.total_sqft_ly) AS previous_sqft,
  ((SUM(f.total_sqft) - SUM(f.total_sqft_ly)) / SUM(f.total_sqft_ly)) * 100 AS yoy_growth_pct
FROM
  logistics_assets.dim_facilities f
WHERE
  f.status = 'ACTIVE'
GROUP BY
  f.region_code, f.facility_type
ORDER BY
  current_sqft DESC;
